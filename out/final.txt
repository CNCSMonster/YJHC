#include <stdlib.h>

}Man;

	self->age=a ;
}

	printf("Hello,I'm %d years old!",self->age) ;
}
int main(){
	Man a ;
	a.age = 11 ;
	_yjhc_Man_selfIntroduce(&a) ;
	_yjhc_Man_setAge(&a,33) ;
